function choose_move(current_positions, other_players_moves):
    # Analyze current positions and potential moves
    possible_moves = [1, 3, 5]
    best_move = None
    max_advancement = 0
    
    for move in possible_moves:
        new_position = current_positions['P1'] + move
        # Check if this move can win
        if new_position >= 23:
            return move
        # Check for potential collisions
        collision = False
        for player in other_players_moves:
            if other_players_moves[player] == move:
                collision = True
                break
        if not collision:
            # Choose the move that maximizes advancement without collision
            if new_position > max_advancement:
                max_advancement = new_position
                best_move = move
    return best_move
